# Type Descriptions

This document contains descriptions and explanations of the different objects seen in ReactDraw

## Point

```typescript
type Point = [number, number];
```

## RectBounds

```typescript
type RectBounds = {
  top: number;
  right: number;
  bottom: number;
  left: number;
};
```

## DrawingContainer

```typescript
type DrawingContainer = {
  div: HTMLDivElement;
  id: string;
};
```

## DrawingData

```typescript
type DrawingData = {
  id: string;
  containerDiv: HTMLDivElement;
  coords: Point[];
  element: HTMLElement | SVGSVGElement | null;
  style: ToolPropertiesMap;
  toolId: string;
  customData: Map<string, any>;
};
```

## OnResizeContext

```typescript
type OnResizeContext = {
  viewContainer: HTMLDivElement;
  previousPoint: Point;
  newPoint: Point;
  shouldPreserveAspectRatio: boolean;
};
```

## SelectMode

```typescript
type SelectMode =
  | "drag"
  | "rotate"
  | "resize-nw"
  | "resize-ne"
  | "resize-se"
  | "resize-sw";
```

## ToolPropertiesMap

```typescript
type ToolPropertiesMap = {
  lineWidth: string;
  zIndex: string;
  color: string;
  fontSize: string;
  background: string;
  opacity: string;
  [id: string]: string;
};
```
